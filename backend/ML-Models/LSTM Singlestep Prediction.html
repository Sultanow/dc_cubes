import numpy as np
# beinhaltet die Vorgerasagen 
test_predictions = []

# die letzten n_input Punkte aus dem Test Satz
first_eval_batch = train[-n_input:] 
# Umformen fr das format des RNN (gleich wie im TimeseriesGenerator)
current_batch = first_eval_batch.reshape((1,n_input,n_features))

# wie lange in die Zukunft soll vorhergesagt werden?
# len(test) in dem fall 36 steps
for i in range (len(test)):
    
    # ein Step voraus ausgehen von den historischen 6 Punkten
    current_pred = model.predict(current_batch)[0]
    
    test_predictions.append(current_pred)
    
    # update current batch sodass er die Vorhersage beinhaltet
    current_batch = np.append(current_batch[:,1:,:],[[test[i]]], axis=1)
